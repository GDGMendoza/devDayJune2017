│    ├──  app.e2e-spec.js

├──  src/

│   │   │

│   │   │   └──  techs.spec.ts
│   │   │
│   │   ├── footer.ts
│   │   ├── footer.spec.ts
│   │   ├── header.ts
│   │   ├── header.spec.ts
│   │   ├── main.ts
│   │   ├── main.spec.ts
│   │   ├── title.ts

│   │
